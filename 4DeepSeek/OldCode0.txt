#include <iostream>
#include <vector>
#include <filesystem>
#include <png++/png.hpp>

// Function to read PNG file and extract channel data
std::vector<uint8_t> readPNGChannel(const std::string &filePath, int colorChannel)
{
    png::image<png::rgba_pixel> image(filePath);
    std::vector<uint8_t> data;
    for (int y = 0; y < image.get_height(); ++y)
    {
        for (int x = 0; x < image.get_width(); ++x)
        {
            uint8_t pixelValue = image[y][x].channels()[colorChannel];
            data.push_back(pixelValue);
        }
    }
    return data;
}

// Function to pack the extracted values into 64-bit format
uint64_t packValues(const std::vector<uint8_t> &values, int bitCount)
{
    uint64_t packedValue = 0;
    for (size_t i = 0; i < values.size(); ++i)
    {
        packedValue |= static_cast<uint64_t>(values[i]) << (i * bitCount);
    }
    return packedValue;
}

int main()
{
    std::vector<std::string> filePaths = {
        "GameFiles/MapCreator/Map/TemplateMaps/HeightMap.png",
        "GameFiles/MapCreator/Map/TemplateMaps/AreaTypeMap.png",
        "GameFiles/MapCreator/Map/TemplateMaps/ResNumMap.png",
        "GameFiles/MapCreator/Map/TemplateMaps/ResIDMap.png",
        "GameFiles/MapCreator/Map/TemplateMaps/CountryMap.png",
        "GameFiles/MapCreator/Map/TemplateMaps/ProvinceMap.png",
        "GameFiles/MapCreator/Map/TemplateMaps/InfrastructureMap.png",
        "GameFiles/MapCreator/Map/TemplateMaps/SoilMap.png",
        "GameFiles/MapCreator/Map/TemplateMaps/ForestMap.png"};

    std::vector<uint64_t> packedData;

    for (size_t i = 0; i < filePaths.size(); ++i)
    {
        std::vector<uint8_t> values = readPNGChannel(filePaths[i], (i == 0) ? 0 : (i == 1)         ? 1
                                                                              : (i >= 2 && i <= 7) ? 2
                                                                                                   : 3);
        uint64_t packedValue = packValues(values, (i == 0) ? 8 : (i == 1)         ? 8
                                                             : (i >= 2 && i <= 5) ? 7
                                                                                  : 4);
        packedData.push_back(packedValue);
    }

    // Output the packed data in CSV format
    std::cout << "x,y,value_hex64" << std::endl;
    for (size_t y = 0; y < 100; ++y)
    {
        for (size_t x = 0; x < 100; ++x)
        {
            size_t index = y * 100 + x;
            std::cout << x << "," << y << "," << std::hex << packedData[index] << std::endl;
        }
    }

    return 0;
}
